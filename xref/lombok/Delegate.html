22  package lombok;
23  
24  import java.lang.annotation.ElementType;
25  import java.lang.annotation.Retention;
26  import java.lang.annotation.RetentionPolicy;
27  import java.lang.annotation.Target;
28  
29  /**
30   * @deprecated Use {@link lombok.experimental.Delegate} instead.
31   */
32  @Target({ElementType.FIELD, ElementType.METHOD})
33  @Retention(RetentionPolicy.SOURCE)
34  @Deprecated
35  public @interface Delegate {
36  	/**
37  	 * Normally the type of the field is used as delegate type. However, to choose a different type to delegate, you can list one (or more) types here. Note that types with
38  	 * type arguments can only be done as a field type. A solution for this is to create a private inner interface/class with the appropriate types extended, and possibly
39  	 * with all methods you'd like to delegate listed, and then supply that class here. The field does not actually have to implement the type you're delegating; the
40  	 * type listed here is used only to determine which delegate methods to generate.
41  	 * 
42  	 * NB: All methods in {@code Object}, as well as {@code canEqual(Object other)} will never be delegated.
43  	 * 
44  	 * @return For each method (not already in {@code java.lang.Object}) in these types, generate a delegate method.
45  	 */
46  	Class<?>[] types() default {};
47  	
48  	/**
49  	 * Each method in any of the types listed here (include supertypes) will <em>not</em> be delegated.
50  	 * 
51  	 * NB: All methods in {@code Object}, as well as {@code canEqual(Object other)} will never be delegated.
52  	 * 
53  	 * @return For each method (not already in {@code java.lang.Object}) in these types, skip generating a delegate method (overrides {@code types()}).
54  	 */
55  	Class<?>[] excludes() default {};
56  }
