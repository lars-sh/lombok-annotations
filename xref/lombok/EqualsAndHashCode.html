22  package lombok;
23  
24  import java.lang.annotation.ElementType;
25  import java.lang.annotation.Retention;
26  import java.lang.annotation.RetentionPolicy;
27  import java.lang.annotation.Target;
28  
29  /**
30   * Generates implementations for the {@code equals} and {@code hashCode} methods inherited by all objects, based on relevant fields.
31   * <p>
32   * Complete documentation is found at <a href="https://projectlombok.org/features/EqualsAndHashCode">the project lombok features page for &#64;EqualsAndHashCode</a>.
33   */
34  @Target(ElementType.TYPE)
35  @Retention(RetentionPolicy.SOURCE)
36  public @interface EqualsAndHashCode {
37  	/**
38  	 * Any fields listed here will not be taken into account in the generated {@code equals} and {@code hashCode} implementations.
39  	 * Mutually exclusive with {@link #of()}.
40  	 * <p>
41  	 * Will soon be marked {@code @Deprecated}; use the {@code @EqualsAndHashCode.Exclude} annotation instead.
42  	 * 
43  	 * @return A list of fields to exclude.
44  	 */
45  	String[] exclude() default {};
46  	
47  	/**
48  	 * If present, explicitly lists the fields that are to be used for identity.
49  	 * Normally, all non-static, non-transient fields are used for identity.
50  	 * <p>
51  	 * Mutually exclusive with {@link #exclude()}.
52  	 * <p>
53  	 * Will soon be marked {@code @Deprecated}; use the {@code @EqualsAndHashCode.Include} annotation together with {@code @EqualsAndHashCode(onlyExplicitlyIncluded = true)}.
54  	 * 
55  	 * @return A list of fields to use (<em>default</em>: all of them).
56  	 */
57  	String[] of() default {};
58  	
59  	/**
60  	 * Call on the superclass's implementations of {@code equals} and {@code hashCode} before calculating for the fields in this class.
61  	 * <strong>default: false</strong>
62  	 * 
63  	 * @return Whether to call the superclass's {@code equals} implementation as part of the generated equals algorithm.
64  	 */
65  	boolean callSuper() default false;
66  	
67  	/**
68  	 * Normally, if getters are available, those are called. To suppress this and let the generated code use the fields directly, set this to {@code true}.
69  	 * <strong>default: false</strong>
70  	 * 
71  	 * @return If {@code true}, always use direct field access instead of calling the getter method.
72  	 */
73  	boolean doNotUseGetters() default false;
74  	
75  	/**
76  	 * Any annotations listed here are put on the generated parameter of {@code equals} and {@code canEqual}.
77  	 * This is useful to add for example a {@code Nullable} annotation.<br>
78  	 * The syntax for this feature depends on JDK version (nothing we can do about that; it's to work around javac bugs).<br>
79  	 * up to JDK7:<br>
80  	 *  {@code @EqualsAndHashCode(onParam=@__({@AnnotationsGoHere}))}<br>
81  	 * from JDK8:<br>
82  	 *  {@code @EqualsAndHashCode(onParam_={@AnnotationsGohere})} // note the underscore after {@code onParam}.
83  	 *  
84  	 * @return List of annotations to apply to the generated parameter in the {@code equals()} method.
85  	 */
86  	AnyAnnotation[] onParam() default {};
87  	
88  	/**
89  	 * Placeholder annotation to enable the placement of annotations on the generated code.
90  	 * @deprecated Don't use this annotation, ever - Read the documentation.
91  	 */
92  	@Deprecated
93  	@Retention(RetentionPolicy.SOURCE)
94  	@Target({})
95  	@interface AnyAnnotation {}
96  	
97  	/**
98  	 * Only include fields and methods explicitly marked with {@code @EqualsAndHashCode.Include}.
99  	 * Normally, all (non-static, non-transient) fields are included by default.
100 	 * 
101 	 * @return If {@code true}, don't include non-static non-transient fields automatically (default: {@code false}).
102 	 */
103 	boolean onlyExplicitlyIncluded() default false;
104 	
105 	/**
106 	 * If present, do not include this field in the generated {@code equals} and {@code hashCode} methods.
107 	 */
108 	@Target(ElementType.FIELD)
109 	@Retention(RetentionPolicy.SOURCE)
110 	public @interface Exclude {}
111 	
112 	/**
113 	 * Configure the behaviour of how this member is treated in the {@code equals} and {@code hashCode} implementation; if on a method, include the method's return value as part of calculating hashCode/equality.
114 	 */
115 	@Target({ElementType.FIELD, ElementType.METHOD})
116 	@Retention(RetentionPolicy.SOURCE)
117 	public @interface Include {
118 		/**
119 		 * Defaults to the method name of the annotated member.
120 		 * If on a method and the name equals the name of a default-included field, this member takes its place.
121 		 * 
122 		 * @return If present, this method serves as replacement for the named field.
123 		 */
124 		String replaces() default "";
125 
126 		/**
127 		 * Higher ranks are considered first. Members of the same rank are considered in the order they appear in the source file.
128 		 * 
129 		 * If not explicitly set, the {@code default} rank for primitives is 1000, and for primitive wrappers 800.
130 		 * 
131 		 * @return ordering within the generating {@code equals} and {@code hashCode} methods; higher numbers are considered first.
132 		 */
133 		int rank() default 0;
134 	}
135 }
