22  package lombok.experimental;
23  
24  import static java.lang.annotation.ElementType.*;
25  import static java.lang.annotation.RetentionPolicy.*;
26  
27  import java.lang.annotation.Retention;
28  import java.lang.annotation.Target;
29  
30  import lombok.Singular;
31  
32  /**
33   * The SuperBuilder annotation creates a so-called 'builder' aspect to the class that is annotated with {@code @SuperBuilder}, but which works well when extending.
34   * It is similar to {@code @Builder}, except it is only legal on types, is less configurable, but allows you to {@code extends} other builder-able classes.
35   * <p>
36   * All classes in the hierarchy must be annotated with {@code @SuperBuilder}.
37   * <p>
38   * Lombok generates 2 inner 'builder' classes, which extend the parent class' builder class (unless your class doesn't have an extends clause).
39   * Lombok also generates a static method named {@code builder()}, and a protected constructor that takes 1 argument of the builderclass type.
40   * <p>
41   * The <code><strong>T</strong>Builder</code> class contains 1 method for each parameter of the annotated
42   * constructor / method (each field, when annotating a class), which returns the builder itself.
43   * The builder also has a <code>build()</code> method which returns a completed instance of the original type.
44   * <p>
45   * Complete documentation is found at <a href="https://projectlombok.org/features/experimental/SuperBuilder">the project lombok features page for &#64;SuperBuilder</a>.
46   *
47   * @see Singular
48   */
49  @Target(TYPE)
50  @Retention(SOURCE)
51  public @interface SuperBuilder {
52  	/** @return Name of the method that creates a new builder instance. Default: {@code builder}. If the empty string, suppress generating the {@code builder} method. */
53  	String builderMethodName() default "builder";
54  
55  	/** @return Name of the method in the builder class that creates an instance of your {@code @Builder}-annotated class. */
56  	String buildMethodName() default "build";
57  
58  	/**
59  	 * If <code>true</code>, generate an instance method to obtain a builder that is initialized with the values of this instance.
60  	 * In this case, all superclasses must also have <code>toBuilder=true</code>.
61  	 *
62  	 * @return Whether to generate a {@code toBuilder()} method.
63  	 */
64  	boolean toBuilder() default false;
65  
66  	/**
67  	 * Prefix to prepend to 'set' methods in the generated builder class. By default, generated methods do not include a prefix.
68  	 *
69  	 * For example, a method normally generated as {@code someField(String someField)} would instead be
70  	 * generated as {@code withSomeField(String someField)} if using {@code @SuperBuilder(setterPrefix = "with")}.
71  	 *
72  	 * Note that using "with" to prefix builder setter methods is strongly discouraged as "with" normally
73  	 * suggests immutable data structures, and builders by definition are mutable objects.
74  	 * 
75  	 * For {@code @Singular} fields, the generated methods are called {@code withName}, {@code withNames}, and {@code clearNames}, instead of
76  	 * the default {@code name}, {@code names}, and {@code clearNames}.
77  	 * 
78  	 * This prefix only applies to the 'set' methods for the fields of the annotated class. 
79  	 * For consistency reasons, you should use the same prefix on all superclasses and subclasses that use {@code @SuperBuilder}.
80  	 * 
81  	 * @return The prefix to prepend to generated method names.
82  	 */
83  	String setterPrefix() default "";
84  }
